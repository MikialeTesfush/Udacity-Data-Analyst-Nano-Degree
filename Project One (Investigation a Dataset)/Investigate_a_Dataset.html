#Creating  a function 
def compute (cols):
    # for the movie with highest budget 
    high=df_movie[cols].idxmax()
    high_infos=pd.DataFrame(df_movie.loc[high])
    # for the movie with lowest budget 
    low=df_movie[cols].idxmin()
    low_infos=pd.DataFrame(df_movie.loc[low])
    #print the movie with high and low Budget
    print(df_movie['original_title'][high],", is the Movie with Highest "+ cols )
    print(df_movie['original_title'][low],", is the Movie with Lowest "+ cols)
    #combine data into one 
    combined_data=pd.concat([high_infos,low_infos],axis=1)
    return combined_data
